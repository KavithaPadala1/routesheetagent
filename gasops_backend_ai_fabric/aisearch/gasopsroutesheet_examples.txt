SQL:
SELECT
  td.WorkDescription,
  COUNT(td.TicketNumber) AS TicketCount
FROM cedemo_RouteSheetHeader h
JOIN cedemo_RouteSheetTicketDetails td ON h.RouteSheetID = td.RouteSheetID
WHERE h.IsActive = 1
  AND td.IsActive = 1
  AND YEAR(h.RouteSheetDate) = 2026
GROUP BY td.WorkDescription
ORDER BY TicketCount DESC;

-- 1. Overview for District category
WITH DistinctData AS (
    SELECT DISTINCT
        h.RouteSheetID,
        rs.EmployeeName,
        h.Region,
        t.WorkDescription,
        t.TicketNumber
    FROM vm_cedemo_routesheetheader h
    JOIN cedemo_RouteSheetTicketDetails t 
        ON h.RouteSheetID = t.RouteSheetID
    JOIN cedemo_RSAssignedEmployeeDetails rs 
        ON t.RouteSheetTicketDetailsID = rs.RouteSheetTicketDetailsID
    WHERE
        h.IsActive = 1
        AND t.IsActive = 1
        AND rs.IsDeleted = 0
        AND h.category = 'District'
        AND rs.JobTitle = 'Supervisor'
        AND CAST(h.RouteSheetDate AS DATE) = '2026-02-05'
)
SELECT
    EmployeeName AS SupervisorName,
    COUNT(DISTINCT WorkDescription) AS WorkDescriptionCount,
    STRING_AGG(WorkDescription, ', ') AS WorkDescriptions,
    COUNT(DISTINCT TicketNumber) AS TotalTicketsinDistrictCategory,
    STRING_AGG(TicketNumber, ', ') AS TicketNumbers,   
    CASE
        WHEN Region = 'X' THEN 'Bronx'
        WHEN Region = 'M' THEN 'Manhattan'
        WHEN Region = 'Q' THEN 'Queens'
        WHEN Region = 'W' THEN 'Westchester'
        ELSE Region
    END AS Region
FROM DistinctData
GROUP BY EmployeeName, Region
ORDER BY EmployeeName, Region;
