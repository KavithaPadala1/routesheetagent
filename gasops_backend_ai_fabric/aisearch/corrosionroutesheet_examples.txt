-- Section 1 — Employee Count by Work Description (DATA)
SELECT WorkDescription, COUNT(DISTINCT ITSID) AS EmployeeCount
FROM cedemo_CorrosionRSAssignedEmployeeDetails ae
JOIN cedemo_CorrosionRouteSheetHeader rh ON ae.CorrosionRouteSheetID = rh.CorrosionRouteSheetID
WHERE rh.IsActive = 1
  AND ae.IsDeleted = 0
  AND CAST(RouteSheetDate AS DATE) = '2026-04-15'
GROUP BY WorkDescription
ORDER BY EmployeeCount DESC;

User : show me routesheets by borough in corrosion routesheet
SQL:
 SELECT
    CASE
        WHEN ae.Borough = 'X' THEN 'Bronx'
        WHEN ae.Borough = 'M' THEN 'Manhattan'
        WHEN ae.Borough = 'Q' THEN 'Queens'
        WHEN ae.Borough = 'W' THEN 'Westchester'
        WHEN ae.Borough = 'BK (Brooklyn)' THEN 'Brooklyn'
        WHEN ae.Borough = 'S (Staten Island)' THEN 'Staten Island'
        ELSE ae.Borough
    END AS Borough,
    h.RouteSheetDate,
    h.RouteSheetDay,
    h.RouteSheetStatus,
    h.Supervisor1Name,
    h.PlannerName
FROM cedemo_CorrosionRouteSheetHeader h
JOIN cedemo_CorrosionRSAssignedEmployeeDetails ae
    ON h.CorrosionRouteSheetID = ae.CorrosionRouteSheetID
WHERE h.IsActive = 1
  AND ae.IsDeleted = 0
  AND YEAR(h.RouteSheetDate) = 2026
ORDER BY Borough ASC, h.RouteSheetDate DESC;
SELECT
    CASE
        WHEN ae.Borough = 'X' THEN 'Bronx'
        WHEN ae.Borough = 'M' THEN 'Manhattan'
        WHEN ae.Borough = 'Q' THEN 'Queens'
        WHEN ae.Borough = 'W' THEN 'Westchester'
        WHEN ae.Borough = 'BK (Brooklyn)' THEN 'Brooklyn'
        WHEN ae.Borough = 'S (Staten Island)' THEN 'Staten Island'
        ELSE ae.Borough
    END AS Borough,
    COUNT(DISTINCT h.CorrosionRouteSheetID) AS RouteSheetCount
FROM cedemo_CorrosionRouteSheetHeader h
JOIN cedemo_CorrosionRSAssignedEmployeeDetails ae
    ON h.CorrosionRouteSheetID = ae.CorrosionRouteSheetID
WHERE h.IsActive = 1
  AND ae.IsDeleted = 0
  AND YEAR(h.RouteSheetDate) = 2026
GROUP BY CASE
        WHEN ae.Borough = 'X' THEN 'Bronx'
        WHEN ae.Borough = 'M' THEN 'Manhattan'
        WHEN ae.Borough = 'Q' THEN 'Queens'
        WHEN ae.Borough = 'W' THEN 'Westchester'
        WHEN ae.Borough = 'BK (Brooklyn)' THEN 'Brooklyn'
        WHEN ae.Borough = 'S (Staten Island)' THEN 'Staten Island'
        ELSE ae.Borough
    END
ORDER BY RouteSheetCount DESC;
