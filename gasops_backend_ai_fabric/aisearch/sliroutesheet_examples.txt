User : Are there any OQ Exceptions in todays contractor SLI route
SQL:
-- contractor employees with OQExceptions
SELECT 
    EmployeeName,
    ITSID,
    Region,
    ThursdayDate,
    ThursdayQualifiedWorkActivityFunctions,
    ThursdayNOTQualifiedWorkActivityFunctions,
    ThursdayMissingCoveredTaskNums
FROM cedemo_ContractorsSLIEmployeeScheduleDetails
WHERE 
    YearNumber = '2026'
    AND ThursdayDate = '2026-04-16'
    AND IsDeleted = 0
    AND (
        ThursdayNOTQualifiedWorkActivityFunctions IS NOT NULL 
        AND ThursdayNOTQualifiedWorkActivityFunctions != '' AND ThursdayMissingCoveredTaskNums IS NOT NULL and ThursdayMissingCoveredTaskNums != ''
    )
ORDER BY EmployeeName ASC;
-- count of contractor employees with OQExceptions
SELECT 
    COUNT(DISTINCT ITSID) AS contractoremployeescountwithOQExceptions
FROM cedemo_ContractorsSLIEmployeeScheduleDetails
WHERE 
    YearNumber = '2026'
    AND ThursdayDate = '2026-04-16'
    AND IsDeleted = 0
    AND (
        ThursdayNOTQualifiedWorkActivityFunctions IS NOT NULL 
        AND ThursdayNOTQualifiedWorkActivityFunctions != '' AND ThursdayMissingCoveredTaskNums IS NOT NULL and ThursdayMissingCoveredTaskNums != ''
    );

User : show me this week shift details for John Smith in company SLI
SQL:
SELECT EmployeeName, ITSID, Region, YearNumber, WeekNumber, SundayShift, MondayShift, TuesdayShift, WednesdayShift, ThursdayShift, FridayShift, SaturdayShift
FROM cedemo_CompanySLIEmployeeScheduleDetails
WHERE IsDeleted = 0
  AND EmployeeName LIKE '%John Smith%'
  AND YearNumber = '2026'
  AND WeekNumber = DATEPART(WEEK, GETDATE())
ORDER BY EmployeeName ASC;
SELECT COUNT(DISTINCT ITSID) AS EmployeeCount
FROM cedemo_CompanySLIEmployeeScheduleDetails
WHERE IsDeleted = 0
  AND EmployeeName LIKE '%John Smith%'
  AND YearNumber = '2026'
  AND WeekNumber = DATEPART(WEEK, GETDATE());
