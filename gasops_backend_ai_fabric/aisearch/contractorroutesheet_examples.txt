-- count of ticket numbers which has secondinspector assigned and not assigned
SELECT
    COUNT(*) AS TotalTicketsCount,
    SUM(CASE WHEN HasInspector = 1 THEN 1 ELSE 0 END) 
        AS TicketsCountWithSecondInspectorAssigned,
    SUM(CASE WHEN HasInspector = 0 THEN 1 ELSE 0 END) 
        AS TicketsCountWithSecondInspectorNotAssigned
FROM (
    SELECT
        TicketNumber,
        MAX(CASE 
            WHEN SecondInspectorNameITSID IS NOT NULL 
                 AND SecondInspectorNameITSID <> '' 
            THEN 1 ELSE 0 
        END) AS HasInspector
    FROM cedemo_ContractorRouteSheetTicketDetails
    WHERE IsActive = 1
      AND Region = 'X'
      AND CAST(RouteSheetDate AS DATE) = '2026-01-12'
    GROUP BY TicketNumber
) t;

SELECT 
    WorkType,
    COUNT(DISTINCT TicketNumber) AS TotalTickets,
    SUM(CASE WHEN RegionName = 'Bronx' THEN 1 ELSE 0 END) AS Bronx,
    SUM(CASE WHEN RegionName = 'Manhattan' THEN 1 ELSE 0 END) AS Manhattan,
    SUM(CASE WHEN RegionName = 'Queens' THEN 1 ELSE 0 END) AS Queens,
    SUM(CASE WHEN RegionName = 'Westchester' THEN 1 ELSE 0 END) AS Westchester
FROM (
    SELECT DISTINCT TicketNumber ,
        WorkType,
        CASE
            WHEN Region = 'X' THEN 'Bronx'
            WHEN Region = 'M' THEN 'Manhattan'
            WHEN Region = 'Q' THEN 'Queens'
            WHEN Region = 'W' THEN 'Westchester'
            ELSE Region
        END AS RegionName
    FROM cedemo_ContractorRouteSheetTicketDetails
    WHERE IsActive = 1

) t
GROUP BY WorkType
ORDER BY TotalTickets DESC;
