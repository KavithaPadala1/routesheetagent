-- Section 1: Employee Count by Work Description
SELECT WorkDescription, COUNT(DISTINCT EmployeeID) as EmployeeCount 
FROM cedemo_TunnelsRSAssignedEmployeeDetails aed
JOIN cedemo_TunnelsRouteSheetHeader h ON aed.TunnelsRouteSheetID = h.TunnelsRouteSheetID
WHERE aed.IsDeleted = 0 AND h.IsActive = 1 
  AND CAST(h.RouteSheetDate AS DATE) = '2026-04-09'
GROUP BY WorkDescription 
ORDER BY EmployeeCount DESC;
