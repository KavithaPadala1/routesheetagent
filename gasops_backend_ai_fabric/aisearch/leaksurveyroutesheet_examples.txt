-- contractor employees with OQExceptions
SELECT 
    EmployeeName,
    ITSID,
    Region,
    ThursdayDate,
    ThursdayQualifiedWorkDescriptions,
    ThursdayNOTQualifiedWorkDescriptions,
    ThursdayMissingCoveredTaskNums
FROM cedemo_ContractorsLeakSurveyEmployeeScheduleDetails
WHERE 
    YearNumber = '2026'
    AND ThursdayDate = '2026-04-16'
    AND IsDeleted = 0
    AND (
        ThursdayNOTQualifiedWorkDescriptions IS NOT NULL 
        AND ThursdayNOTQualifiedWorkDescriptions != '' AND ThursdayMissingCoveredTaskNums IS NOT NULL and ThursdayMissingCoveredTaskNums != ''
    )
ORDER BY EmployeeName ASC;
-- count of contractor employees with OQExceptions
SELECT 
    COUNT(DISTINCT ITSID) AS contractoremployeescountwithOQExceptions
FROM cedemo_ContractorsLeakSurveyEmployeeScheduleDetails
WHERE 
    YearNumber = '2026'
    AND ThursdayDate = '2026-04-16'
    AND IsDeleted = 0
    AND (
        ThursdayNOTQualifiedWorkDescriptions IS NOT NULL 
        AND ThursdayNOTQualifiedWorkDescriptions != '' AND ThursdayMissingCoveredTaskNums IS NOT NULL and ThursdayMissingCoveredTaskNums != ''
    );
